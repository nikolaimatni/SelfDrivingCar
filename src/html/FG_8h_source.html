124  virtual ADVec DynamicsF(int t, const ADVec &xt, const ADVec &ut) override {

127  AD<double> psi = xt[PSI];
128  AD<double> v = xt[V];
129  AD<double> cte = xt[CTE];

131  AD<double> a = ut[A];
132  AD<double> delta = ut[DELTA];
133 
134  AD<double> f = coeffs_[0] + coeffs_[1] * x + coeffs_[2] * CppAD::pow(x, 2) +
135  coeffs_[3] * CppAD::pow(x, 3);
136 
137  AD<double> psides = CppAD::atan(coeffs_[1] + 2 * coeffs_[2] * x +
138  3 * coeffs_[3] * CppAD::pow(x, 2));
139 
140  ADVec fxtut(nx());
141  fxtut[X] = x + v * CppAD::cos(psi) * dt_;
142  fxtut[Y] = y + v * CppAD::sin(psi) * dt_;
143  fxtut[PSI] = psi - v / Lf_ * delta * dt_;
144  fxtut[V] = v + a * dt_;
145  fxtut[CTE] = (f - y) + v * CppAD::sin(epsi) * dt_;
146  fxtut[EPSI] = (psi - psides) - v / Lf_ * delta * dt_;
147 
148  return fxtut;
149  }
